	public static  String ReplaceBlank(char string[], int length) {
		if (string == null || length <= 0 )	{
			// 尝试看看能不能这样写
			return null;
		}
		int originalLength = 0;
		int numberOfBlank = 0;
		int i = 0;
		while (string[i] != '\0') {
			// 统计原始数组的长度为多少
			++originalLength;
			// 统计有多少个空格
			if (string[i] == ' ')  {
				++numberOfBlank;
				// ++i;
			}
			++i;
		}
		// System.out.println(string[17]);
		/* newlength 为把空格替换成‘%20’之后的长度 */
		// 虽然最后的newLength值虽然为17，但是实际数组长度大小为18
		int newLength = originalLength + numberOfBlank * 2 ;
		// 这里 newLength 代表的时序号数
		// length - 1 此时也代表着序号数。
		if (newLength < length - 1) {
			// 同样尝试能不能这样写
			return null;    // c++里的意思是：不带返回值的return 语句
		}
		
		int indexOfOriginal = originalLength ;
		int indexOfNew = newLength;
		// 当“前指针（indexOfOriginal)”大于0，同时“后指针(indexOfNew)” 在“前指针”的后面时，执行循环
		while (indexOfOriginal >= 0 && indexOfNew > indexOfOriginal )  {
			// 当有空格时，执行替换
			if (string[indexOfOriginal] == ' ')  {
				string[indexOfNew--] = '0';
				string[indexOfNew--] = '2';
				string[indexOfNew--] = '%';
			}
			// 当没有空格时，执行移动操作
			else  {
				string[indexOfNew-- ] = string[indexOfOriginal];
			}
			// 总之每次结尾时，“前指针”执行一次前移操作
			--indexOfOriginal;
		}
		// 虽然最后结果,result 都是字符串类型，但是:   
		// String result = String.valueOf(string);    // 通过String 类调用的方法：
																		// 最后显示结果只有字符串，没有中括号和逗号，更像字符串。
		 String result = Arrays.toString(string) ;   // 通过Arrays类调用的方法：
																	// 最后结果有中括号和逗号 。同时要导入包java.util.* 。否则报错找不到变量Arrays!
		return result;
	}
